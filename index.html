MinorWork

The PHP Framework you can understand and play with.

composer require minorwork/minorwork

 9   'loginAct'  => ['POST', '/login',  '\Login:login'],
10   'article'   => [        '/a/{id}', '\Article:show'],
11   'users'     => [        '/u',      '\User:list'],
12   'userPage'  => [        '/u/{id}', '\User:profile'],
13 ]);
14 
15 $app->run();
